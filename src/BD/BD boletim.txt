CREATE DATABASE boletim;

USE boletim;

CREATE TABLE pacientes(
	cod_boletim INT AUTO_INCREMENT,
	cod_paciente INT,
	nome_paciente VARCHAR(90),
	leito VARCHAR(50),
	estado_geral VARCHAR(50),
	nivel_consciencia VARCHAR(50),
	diurese VARCHAR(50),
	temperatura VARCHAR(50),
	pressao VARCHAR(50),
	respiracao VARCHAR(50),
	data_ultima_atualizacao DATE,
	hora_ultima_atualizacao TIME,
	PRIMARY KEY (cod_boletim));

CREATE TABLE perfil_usuario(
	Cod_Perfil INT AUTO_INCREMENT,
	Desc_Perfil VARCHAR(90),
	PRIMARY KEY(Cod_Perfil));
